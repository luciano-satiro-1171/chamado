# condifura pela primeira vez o git e github no seu PC:

# atualizar repositorios e pacotes:
sudo apt-get -y update
sudo apt-get -y upgrade

# instalar o git:
sudo apt-get -y install git

# configuracoes globais:
git config --global user.name "Luciano dos Santos"
git config --global user.email "luciano.satiro1171@gmail.com"

# crie um chave ssh:
ssh-keygen -t ed25519 -C "luciano.satiro1171@gmail.com"
    Enter
    Enter
    Enter

# inicie o agente ssh:
eval "$(ssh-agent -s)"

# adicione a nova chave:
ssh-add ~/.ssh/id_ed25519

# imprimir na tela a chave e copie:
cat ~/.ssh/id_ed25519.pub

# vah para o github, faca login na sua conta, e siga estes passos:
1. clique na sua foto de perfil no canto superior direito e vah para Settings.
2. no menu ah esquerda, clique em SSH and GPG keys.
3. clique no botao New SSH key.
4. deh um titulo para a chave (por exemplo, "Meu PC Linux") e cole a chave no campo "Key".
5. clique em Add SSH key.

# testar a conexao:
ssh -T git@github.com
# voce deve ver uma mensagem como esta, indicando que a autenticacao foi bem-sucedida:
Hi username! You've successfully authenticated, but GitHub does not provide shell access.

# crie um diretorio para centralizar os projetos:
mkdir ../<crie_um_diretorio>

# acesse o diretorio que voce criou:
cd ../<diretorio_que_foi_criado>

# agora voce pode clonar repositorios do github usando o comando (git clone):
# acesse o projeto no github, clique em ( <>Code ) com a seta para baixo, clique em SSH e copie...
git clone git@github.com:username/repository.git

# acesse o projeto que voce baixou:
cd ../<diretorio_que_foi_criado>
